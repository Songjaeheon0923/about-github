Git Commands Guide

Git 사용법

기본 명령어

Git 설정

$ git config --global user.name “<닉네임>”
# 유저 이름 설정

$ git config --global user.email “<메일 주소>”
# 유저 메일 주소 설정

파일 및 폴더 확인

$ ls
# 폴더 내의 가시 파일 정렬

$ ls -al
# 폴더 내의 숨김 폴더까지 함께 보여줌

Git 저장소 초기화

$ git init
# 깃 저장소 초기화

파일 생성

$ touch <파일명>.<파일 형식>
# 해당 레포지토리에 파일 생성

Git 상태 확인

$ git status
# 폴더 내 파일들의 현재 상태를 확인 가능

파일 스테이징

$ git add <파일명>.<파일 형식>
# 깃이 관리하는 가상 영역에 해당 파일을 전달

$ git add .
# 현재 워킹 디렉토리에 있는 모든 untracked file을 전부 올린다.

커밋

$ git commit
# 현재 add된 파일들을 전부 커밋함
# shift+i로 코멘트 작성 후 esc + :wq로 저장

$ git commit -m “<커밋 메시지>”
# 바로 커밋 메시지 작성

커밋 이력 확인

$ git log
# 현재까지 commit 이력을 보여줌

$ git log -p
# 로그의 변경 사항까지 함께 출력됨 (exit하려면 q)

$ git log -<숫자>
# 숫자만큼의 최신 로그를 보여줌

$ git log --oneline
# 각 로그를 한 줄씩만 간단하게 보여줌

파일 무시 설정

$ nano .gitignore
# 숨기고 싶은 파일 작성 후 ^X로 저장
$ git status
# .gitignore 파일이 untracked file로 나옴
$ git add .gitignore
# .gitignore 파일 커밋

Reset과 Revert

Reset

$ git reset <파일명>
# 해당 파일을 git의 가상 공간에서 untracked로 옮김

$ git reset <해시 7자리> --<soft|mixed|hard>
# soft: 리포지토리만 바꿈
# mixed: working directory만 냅둠
# hard: 리포지토리, staging area, working directory 전부 변경

Revert

$ git revert <해시 7자리>
# 커밋 수정 사항을 되돌렸다는 내용까지 log에 남김

브랜치 관리

브랜치 생성 및 이동

$ git branch
# 현재 가지고 있는 브랜치 목록을 보여줌

$ git branch <새로운 브랜치 이름>
# 새로운 브랜치 생성

$ git checkout <브랜치 이름>
# 해당 브랜치로 이동

브랜치 병합 및 삭제

$ git merge <브랜치 이름>
# 현재 브랜치와 해당 브랜치를 병합

$ git branch -d <브랜치 이름>
# 해당 브랜치를 삭제

원격 저장소

원격 저장소 연결

$ git remote add <저장소 이름> <원격 저장소 주소>
# 원격 저장소 연결

$ git remote -v
# 현재 연결된 원격 저장소를 출력

원격 저장소 작업

$ git push -u <저장소 이름> <브랜치 이름>
# 원격 저장소로 push

$ git clone <원격 저장소 주소>
# 원격 저장소 내용 가져오기

$ git pull
# 원격 저장소 내용 병합

참고 명령어

$ cd <폴더명>
# 해당 폴더로 이동

$ cd ..
# 상위 폴더로 이동
